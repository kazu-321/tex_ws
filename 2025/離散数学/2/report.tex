\section*{(a)}
\begin{flalign*}
  x &= \#(A\cap I\cap N) &&\\
  \#(A\cup I\cup N) &= 60 - 8 = 52 &&\\
  &= 25 + 26 + 26 - 9 - 11 - 8 + x  &&\\
  x &= 3
\end{flalign*}
