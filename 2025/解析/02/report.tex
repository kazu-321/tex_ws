\section*{02-01}
\section*{$(1) \int x\sin 2x dx$}
\begin{flalign*}
  &= \int x\left\{-\frac12\cos 2x\right\}' dx &&\\
  &= -\frac12x\cos 2x - \int -\frac12\cos 2x dx &&\\
  &= -\frac12x\cos 2x + \frac14\sin 2x + C &&
\end{flalign*}

\section*{$(2) \int x\cos 3x dx$}
\begin{flalign*}
  &= \int x\frac13\{\sin 3x\}' dx &&\\
  &= \frac13x\sin 3x - \int \frac13\sin 3x dx &&\\
  &= \frac13x\sin 3x + \frac19\cos 3x + C &&
\end{flalign*}

\section*{$(3) \int \frac{x}{\cos ^2 x} dx$}
\begin{flalign*}
  &= \int x \{\tan x\}' dx &&\\
  &= x\tan x - \int \tan x dx &&\\
  &= x\tan x - \int \frac{\sin x}{\cos x} dx &&\\
  &= x\tan x - \int \frac{-\{\cos x\}'}{\cos x} dx &&\\
  &= x\tan x - (-\log|\cos x|) + C &&\\
  &= x\tan x + \log|\cos x| + C &&
\end{flalign*}

\section*{$(4) \int x\log(x-2) dx$}
\begin{flalign*}
  &= \int \frac12(x^2-2^2)'\log(x-2) dx&&\\
  &= \frac12(x^2-2^2)\log(x-2) - \int \frac12(x^2-2^2)\frac{1}{x-2} dx &&\\
  &= \frac12(x^2-2^2)\log(x-2) - \int \frac12\frac{(x+2)(x-2)}{x-2} dx &&\\
  &= \frac12(x^2-2^2)\log(x-2) - \int \frac12(x+2) dx &&\\
  &= \frac12(x^2-2^2)\log(x-2) - (\frac14x^2 + x) + C
\end{flalign*}

\section*{$(5) \int x^3\log x dx$}
\begin{flalign*}
  &= \int \frac14(x^4)'\log xdx &&\\
  &= \frac14x^4\log x - \int \frac14x^4\frac1x dx &&\\
  &= \frac14x^4\log x - \int \frac14x^3 dx &&\\
  &= \frac14x^4\log x - \frac{1}{16}x^4 +C
\end{flalign*}

\section*{$(6) \int (2x-1)e^x dx$}
\begin{flalign*}
  &= \int (2x-1)(e^x)' dx &&\\
  &= (2x-1)e^x - \int 2e^x dx &&\\
  &= (2x-1)e^x - 2e^x + C &&\\
  &= (2x-3)e^x +C
\end{flalign*}

\section*{$(7) \int xe^{-3x}dx$}
\begin{flalign*}
  &= \int -\frac13 x (e^{-3x})' dx &\\
  &= -\frac13xe^{-3x} - \int -\frac13e^{-3x} dx &&\\
  &= -\frac13xe^{-3x}-\frac19e^{-3x} + C &&\\
  &= e^{-3x}(-\frac13x -\frac19)+C &&
\end{flalign*}

\section*{$(8) \int \log(x+2) dx$}
\begin{flalign*}
  &= \int (x+2)'\log(x+2) dx &&\\
  &= (x+2)\log(x+2) - \int (x+2)\frac{1}{x+2} dx &&\\
  &= (x+2)\log(x+2) - \int 1 dx &&\\
  &= (x+2)\log(x+2) -x + C
\end{flalign*}

\section*{$(9) \int \log(1-x)dx$}
\begin{flalign*}
  &= \int -(1-x)'\log(1-x)dx &&\\
  &= -(1-x)\log(1-x) - \int -(1-x)\frac{1}{1-x} dx &&\\
  &= -(1-x)\log(1-x) - \int -1 dx &&\\
  &= -(1-x)\log(1-x) + x +C&&
\end{flalign*}

\section*{$(1) \int xe^xdx$}
\begin{flalign*}
  &= \int x(e^x)'dx &&\\
  &= xe^x - \int e^x dx &&\\
  &= xe^x - e^x + C &&\\
  &= e^x(x-1) +C
\end{flalign*}

\begin{flalign*}

  &= -\frac12xe^{-2x} - \int -\frac12e^{-2x} dx &&\\
  &= -\frac12xe^{-2x} - \frac14e^{-2x} +C &&\\
  &= e^{-2x}(-\frac12x - \frac14) +C
\end{flalign*}

\section*{$(3) \int (2x-1)e^{2x}dx$}

  &= (x-\frac12)e^{2x} - \frac12e^{2x} +C&&\\
  &= e^{2x}(x-1) +C
\end{flalign*}

\end{flalign*}

  &= -x^2\cos x - \int -2x\cos x dx &&\\
  &= -x^2\cos x - \int -2x(\sin x)' dx &&\\
  &= -x^2\cos x - (-2x\sin x - \int -2\sin x dx) &&\\
  &= -x^2\cos x +x2\sin x+2\cos x + C
\end{flalign*}

  &= -x^2e^{-x} - (2xe^{-x} - \int 2e^{-x}dx)&&\\
  &= -x^2e^{-x} - 2xe^{-x} - 2e^{-x} +C
\end{flalign*}
